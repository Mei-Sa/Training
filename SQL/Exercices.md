
### **Exercice 6**

6.1
```sql
SELECT CDNUMDEP, COUNT(CDNUMCL) AS NB_CLI_DEP FROM TCLIENT 
       GROUP BY CDNUMDEP HAVING COUNT(CDNUMCLI) > (SELECT COUNT(CDNUMDEP)/COUNT(*) FROM TCLIENT) 
       ORDER BY NB_CLI_DEP DESC ;
```
6.2
```sql
SELECT MIN(MTCAPCLI), MAX(MTCAPCLI), AVG(MTCAPCLI), COUNT(LBNOMCLI), CDNUMDEP FROM TCLIENT
       WHERE CDNUMDEP IN (SELECT CDNUMDEP FROM TREGION,TDEPART WHERE TDEPART.CDNUMREG = TREGION.CDNUMREG 
       AND LBNOMREG NOT LIKE '%ILE DE FRANCE%')  
       GROUP BY CDNUMDEP HAVING COUNT(CDNUMCLI) >= 3 ;
```
6.3
```sql
SELECT TARTICL.CDNUMART, LBNOMART FROM TARTICL
       WHERE NOT EXISTS (SELECT TLIGCOM.CDNUMART FROM TLIGCOM WHERE TARTICL.CDNUMART = TLIGCOM.CDNUMART)
       ORDER BY TARTICL.CDNUMART ;
```
6.4
```sql
SELECT TCOMMAN.CDNUMCOM FROM TCOMMAN, TLIGCOM, TARTICL
       WHERE TCOMMAN.CDNUMFAC IS NULL AND NOT EXISTS (SELECT TLIGCOM.CDNUMCOM FROM TLIGCOM, TARTICL 
             WHERE QTARTLCO > QTSTKART AND TLIGCOM.CDNUMART = TARTICL.CDNUMART AND TLIGCOM.CDNUMCOM = TCOMMAN.CDNUMCOM)
       ORDER BY TCOMMAN.CDNUMCOM ;
```
